```tsx preview
  const vseedConfig = {
  "chartType": "column",
  "dataset": [
    {
      "date": "2019",
      "profit": 10,
      "sales": 20,
      "count": 15
    },
    {
      "date": "2020",
      "profit": 30,
      "sales": 60,
      "count": 33
    },
    {
      "date": "2021",
      "profit": 30,
      "sales": 60,
      "count": 22
    },
    {
      "date": "2022",
      "profit": 50,
      "sales": 100,
      "count": 14
    },
    {
      "date": "2023",
      "profit": 40,
      "sales": 80,
      "count": 30
    }
  ],
  "dimensions": [
    {
      "id": "date",
      "alias": "Datum"
    }
  ],
  "measures": [
    {
      "id": "count",
      "alias": "Menge"
    },
    {
      "id": "sales",
      "alias": "Umsatz"
    },
    {
      "id": "profit",
      "alias": "Gewinn"
    }
  ],
  "sortLegend": {
    "order": "desc"
  }
}

  return <VSeedRender vseed={vseedConfig} />
}
```
