# Minimalist

import { useDark, useLang } from '@rspress/core/runtime'
import { APP } from 'minimalist'

export const MinimalistPractice = () => {
  const dark = useDark()
  const lang = useLang()
  const locale = lang === 'en-US' ? 'en-US' : 'zh-CN'
  const theme = dark ? 'dark' : 'light'

  return (
    <div style={{ width: '100%', height: '700px' }}>
      <APP hideLocale hideTheme locale={locale} theme={theme} />
    </div>
  )
}

<MinimalistPractice />
