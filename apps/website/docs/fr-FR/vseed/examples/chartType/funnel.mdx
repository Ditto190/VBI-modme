# Funnel

## Basic Funnel

A funnel chart showing the progressive reduction of data through stages.

```tsx preview
import { VSeedRender } from '@components'

export default () => {
  const vseedConfig = {
  "chartType": "funnel",
  "dataset": [
    {
      "date": "2022",
      "profit": 100,
      "sales": 100
    },
    {
      "date": "2023",
      "profit": 80,
      "sales": 80
    },
    {
      "date": "2020",
      "profit": 60,
      "sales": 60
    },
    {
      "date": "2021",
      "profit": 40,
      "sales": 40
    },
    {
      "date": "2019",
      "profit": 20,
      "sales": 20
    }
  ]
}

  return <VSeedRender vseed={vseedConfig} />
}
```

## Graphique en entonnoir multi-indicateurs

Graphique en entonnoir multi-indicateurs

```tsx preview
import { VSeedRender } from '@components'

export default () => {
  const vseedConfig = {
  "chartType": "funnel",
  "dataset": [
    {
      "profit": 10,
      "sales": 20,
      "profit1": 0.1
    }
  ],
  "measures": [
    {
      "id": "sales",
      "alias": "Ventes"
    },
    {
      "id": "profit",
      "alias": "Bénéfice"
    },
    {
      "id": "profit1",
      "alias": "Taux de rendement"
    }
  ]
}

  return <VSeedRender vseed={vseedConfig} />
}
```

## Graphique en entonnoir combiné

Graphique en entonnoir combiné

```tsx preview
import { VSeedRender } from '@components'

export default () => {
  const vseedConfig = {
  "chartType": "funnel",
  "dataset": [
    {
      "date": "2022",
      "profit": 100,
      "sales": 134
    },
    {
      "date": "2023",
      "profit": 80,
      "sales": 90
    },
    {
      "date": "2020",
      "profit": 60,
      "sales": 70
    },
    {
      "date": "2021",
      "profit": 40,
      "sales": 60
    },
    {
      "date": "2019",
      "profit": 20,
      "sales": 30
    }
  ],
  "measures": [
    {
      "id": "sales",
      "alias": "Ventes",
      "parentId": "left"
    },
    {
      "id": "profit",
      "alias": "Bénéfice",
      "parentId": "right"
    }
  ],
  "dimensions": [
    {
      "id": "date",
      "alias": "Date"
    }
  ]
}

  return <VSeedRender vseed={vseedConfig} />
}
```
