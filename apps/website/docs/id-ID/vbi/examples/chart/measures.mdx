## add-measure

Tambahkan measure


```tsx preview
import { VBI, VBIChartBuilder } from '@visactor/vbi'
import { DEMO_CONNECTOR_ID, VSeedRender } from '@components'
import { useEffect, useState } from 'react'

export default () => {
  const [result, setResult] = useState<any>(null)

  useEffect(() => {
    const run = async () => {
      const builder = VBI.chart.create({
        ...{
          "connectorId": "demoSupermarket",
          "chartType": "table",
          "dimensions": [],
          "measures": [],
          "whereFilter": {
            "id": "root",
            "op": "and",
            "conditions": []
          },
          "havingFilter": {
            "id": "root",
            "op": "and",
            "conditions": []
          },
          "theme": "light",
          "locale": "en-US",
          "version": 1,
          "limit": 20
        },
        connectorId: DEMO_CONNECTOR_ID,
      })
      const applyBuilder = (builder: VBIChartBuilder) => {
  builder.measures.add('sales', node => {
    node.setAlias('Penjualan Awal')
  })
  const measureId = builder.measures.find(node => node.getField() === 'sales')?.getId()
  if (measureId) {
    builder.measures.update(measureId, node => {
      node.setAlias('Penjualan').setAggregate({ func: 'sum' })
    })
  }
}
      applyBuilder(builder)
      setResult(await builder.buildVSeed())
    }
    run()
  }, [])

  if (!result) return <div>Loading...</div>

  return <VSeedRender vseed={result} />
}
```

## measure-with-custom-and-auto-format

Format measure: penjualan memakai format kustom satuan 10 ribu (prefiks ¥, dua desimal), laba memakai format otomatis, dan diskon memakai format persen


```tsx preview
import { VBI, VBIChartBuilder } from '@visactor/vbi'
import { DEMO_CONNECTOR_ID, VSeedRender } from '@components'
import { useEffect, useState } from 'react'

export default () => {
  const [result, setResult] = useState<any>(null)

  useEffect(() => {
    const run = async () => {
      const builder = VBI.chart.create({
        ...{
          "connectorId": "demoSupermarket",
          "chartType": "table",
          "dimensions": [
            {
              "field": "product_type",
              "alias": "Kategori"
            }
          ],
          "measures": [],
          "whereFilter": {
            "id": "root",
            "op": "and",
            "conditions": []
          },
          "havingFilter": {
            "id": "root",
            "op": "and",
            "conditions": []
          },
          "theme": "light",
          "locale": "en-US",
          "version": 1,
          "limit": 20
        },
        connectorId: DEMO_CONNECTOR_ID,
      })
      const applyBuilder = (builder: VBIChartBuilder) => {
  builder.measures
    .add('sales', node => {
      node.setAlias('Penjualan (10K)')
        .setAggregate({ func: 'sum' })
        .setEncoding('column')
        .setFormat({
          type: 'number',
          ratio: 10000,
          symbol: '10K',
          prefix: '¥',
          fractionDigits: 2
        });
    })
    .add('profit', node => {
      node.setAlias('Laba')
        .setAggregate({ func: 'sum' })
        .setEncoding('column')
        .setFormat({ autoFormat: true });
    })
    .add('discount', node => {
      node.setAlias('Diskon Rata-rata')
        .setAggregate({ func: 'avg' })
        .setEncoding('column')
        .setFormat({
          type: 'percent',
          fractionDigits: 1
        });
    });
}
      applyBuilder(builder)
      setResult(await builder.buildVSeed())
    }
    run()
  }, [])

  if (!result) return <div>Loading...</div>

  return <VSeedRender vseed={result} />
}
```

## remove-measure

Hapus measure


```tsx preview
import { VBI, VBIChartBuilder } from '@visactor/vbi'
import { DEMO_CONNECTOR_ID, VSeedRender } from '@components'
import { useEffect, useState } from 'react'

export default () => {
  const [result, setResult] = useState<any>(null)

        ...{
          "connectorId": "demoSupermarket",
          "chartType": "table",
          "dimensions": [],
          "measures": [
            {
              "field": "sales",
              "alias": "Penjualan",
              "aggregate": {
                "func": "sum"
              },
              "encoding": "yAxis"
            },
            {
              "field": "profit",
              "alias": "Laba",
              "aggregate": {
                "func": "sum"
              },
              "encoding": "yAxis"

          ],
          "whereFilter": {
            "id": "root",
            "op": "and",

          },

          "version": 1,
          "limit": 20
```

## update-measure

Perbarui measure


```tsx preview
import { VBI, VBIChartBuilder } from '@visactor/vbi'
import { DEMO_CONNECTOR_ID, VSeedRender } from '@components'
import { useEffect, useState } from 'react'

export default () => {
  const [result, setResult] = useState<any>(null)

  useEffect(() => {
    const run = async () => {
      const builder = VBI.chart.create({
        ...{
          "connectorId": "demoSupermarket",
          "chartType": "table",
          "dimensions": [],
          "measures": [
            {
              "field": "sales",
              "alias": "Penjualan Awal",
              "aggregate": {
                "func": "sum"
              },
              "encoding": "yAxis"
            }
          ],
          "whereFilter": {
            "id": "root",
            "op": "and",
            "conditions": []
          },
          "havingFilter": {
            "id": "root",
            "op": "and",
            "conditions": []
          },
          "theme": "light",
          "locale": "en-US",
          "version": 1,
          "limit": 20

    if (measure) {
      measure.setAlias('Penjualan yang Diubah').setEncoding('yAxis')
```
