## Nama Metrik sebagai Dimensi Hierarki + Satu Metrik

  const vseedConfig = {
  "chartType": "hierarchySankey",
  "dataset": [
    {
      "region": "Cina Utara",
      "metricName": "Penjualan",
      "value": 42
    },
    {
      "region": "Cina Utara",
      "metricName": "Laba",
      "value": 12
    },
    {
      "region": "Cina Timur",
      "metricName": "Penjualan",
      "value": 36

    {
      "region": "Cina Timur",
      "metricName": "Laba",
      "value": 10
    },
    {
      "region": "Cina Selatan",

      "value": 28
    },
    {
      "region": "Cina Selatan",
      "metricName": "Laba",

    {
      "id": "region",
