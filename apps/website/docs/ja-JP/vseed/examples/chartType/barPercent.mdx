# BarPercent

## Basic BarPercent

A percentage bar chart showing the relative proportion of each category.

```tsx preview
import { VSeedRender } from '@components'

export default () => {
  const vseedConfig = {
  "chartType": "barPercent",
  "dataset": [
    {
      "date": "2019",
      "profit": 10,
      "sales": 20
    },
    {
      "date": "2020",
      "profit": 30,
      "sales": 60
    },
    {
      "date": "2021",
      "profit": 30,
      "sales": 60
    },
    {
      "date": "2022",
      "profit": 50,
      "sales": 100
    },
    {
      "date": "2023",
      "profit": 40,
      "sales": 80
    }
  ]
}

  return <VSeedRender vseed={vseedConfig} />
}
```

## 複合百分率横棒グラフ

複合百分率横棒グラフ

```tsx preview
import { VSeedRender } from '@components'

export default () => {
  const vseedConfig = {
  "chartType": "barPercent",
  "measures": [
    {
      "id": "sales",
      "alias": "sales",
      "parentId": "sales-and-profit"
    },
    {
      "id": "profit",
      "alias": "利益",
      "parentId": "sales-and-profit"
    },
    {
      "id": "rateOfReturn",
      "alias": "収益率",
      "parentId": "ratio"
    }
  ],
  "dataset": [
    {
      "date": "2019",
      "profit": 10,
      "sales": 20,
      "rateOfReturn": 0.1
    },
    {
      "date": "2020",
      "profit": 20,
      "sales": 40,
      "rateOfReturn": 0.2
    },
    {
      "date": "2021",
      "profit": 30,
      "sales": 60,
      "rateOfReturn": 0.3
    },
    {
      "date": "2022",
      "profit": 40,
      "sales": 80,
      "rateOfReturn": 0.4
    },
    {
      "date": "2023",
      "profit": 50,
      "sales": 100,
      "rateOfReturn": 0.5
    }
  ],
  "dimensions": [
    {
      "id": "date",
      "alias": "日付"
    }
  ]
}

  return <VSeedRender vseed={vseedConfig} />
}
```
