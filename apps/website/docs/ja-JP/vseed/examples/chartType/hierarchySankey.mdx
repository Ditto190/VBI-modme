## 指標名を階層ディメンションとして使用 + 単一指標

  const vseedConfig = {
  "chartType": "hierarchySankey",
  "dataset": [
    {
      "region": "華北",
      "metricName": "売上",
      "value": 42
    },
    {
      "region": "華北",
      "metricName": "利益",
      "value": 12
    },
    {
      "region": "華東",
      "metricName": "売上",
      "value": 36

      "region": "華東",
      "metricName": "利益",
      "value": 10
    },
    {
      "region": "華南",

      "value": 28
    },
    {
      "region": "華南",

  "dimensions": [
    {
      "id": "region",
