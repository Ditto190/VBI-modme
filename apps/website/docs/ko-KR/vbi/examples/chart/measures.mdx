## add-measure

메저 추가


```tsx preview
import { VBI, VBIChartBuilder } from '@visactor/vbi'
import { DEMO_CONNECTOR_ID, VSeedRender } from '@components'
import { useEffect, useState } from 'react'

export default () => {
  const [result, setResult] = useState<any>(null)

  useEffect(() => {
    const run = async () => {
      const builder = VBI.chart.create({
        ...{
          "connectorId": "demoSupermarket",
          "chartType": "table",
          "dimensions": [],
          "measures": [],
          "whereFilter": {
            "id": "root",
            "op": "and",
            "conditions": []
          },
          "havingFilter": {
            "id": "root",
            "op": "and",
            "conditions": []
          },
          "theme": "light",
          "locale": "en-US",
          "version": 1,
          "limit": 20
        },
        connectorId: DEMO_CONNECTOR_ID,
      })
      const applyBuilder = (builder: VBIChartBuilder) => {
  builder.measures.add('sales', node => {
    node.setAlias('원래 매출')
  })
  const measureId = builder.measures.find(node => node.getField() === 'sales')?.getId()
  if (measureId) {
    builder.measures.update(measureId, node => {
      node.setAlias('매출').setAggregate({ func: 'sum' })
    })
  }
}
      applyBuilder(builder)
      setResult(await builder.buildVSeed())
    }
    run()
  }, [])

  if (!result) return <div>Loading...</div>

  return <VSeedRender vseed={result} />
}
```

## measure-with-custom-and-auto-format

메저 서식: 매출은 만 단위 사용자 지정 서식(¥ 접두사, 소수 2자리), 이익은 자동 서식, 할인은 퍼센트 서식 사용


```tsx preview
import { VBI, VBIChartBuilder } from '@visactor/vbi'
import { DEMO_CONNECTOR_ID, VSeedRender } from '@components'
import { useEffect, useState } from 'react'

export default () => {
  const [result, setResult] = useState<any>(null)

  useEffect(() => {
    const run = async () => {
      const builder = VBI.chart.create({
        ...{
          "connectorId": "demoSupermarket",
          "chartType": "table",
          "dimensions": [
            {
              "field": "product_type",
              "alias": "카테고리"
            }
          ],
          "measures": [],
          "whereFilter": {
            "id": "root",
            "op": "and",
            "conditions": []
          },
          "havingFilter": {
            "id": "root",
            "op": "and",
            "conditions": []
          },
          "theme": "light",
          "locale": "en-US",
          "version": 1,
          "limit": 20
        },
        connectorId: DEMO_CONNECTOR_ID,
      })
      const applyBuilder = (builder: VBIChartBuilder) => {
  builder.measures
    .add('sales', node => {
      node.setAlias('매출(만)')
        .setAggregate({ func: 'sum' })
        .setEncoding('column')
        .setFormat({
          type: 'number',
          ratio: 10000,
          symbol: '만',
          prefix: '¥',
          fractionDigits: 2
        });
    })
    .add('profit', node => {
      node.setAlias('이익')
        .setAggregate({ func: 'sum' })
        .setEncoding('column')
        .setFormat({ autoFormat: true });
    })
    .add('discount', node => {
      node.setAlias('평균 할인')
        .setAggregate({ func: 'avg' })
        .setEncoding('column')
        .setFormat({
          type: 'percent',
          fractionDigits: 1
        });
    });
}
      applyBuilder(builder)
      setResult(await builder.buildVSeed())
    }
    run()
  }, [])

  if (!result) return <div>Loading...</div>

  return <VSeedRender vseed={result} />
}
```

## remove-measure

```tsx preview
import { VBI, VBIChartBuilder } from '@visactor/vbi'
import { DEMO_CONNECTOR_ID, VSeedRender } from '@components'

            {
              "field": "sales",
              "alias": "매출",
              "aggregate": {
                "func": "sum"
              },
              "encoding": "yAxis"

              "field": "profit",
```

## update-measure

메저 업데이트


```tsx preview
import { VBI, VBIChartBuilder } from '@visactor/vbi'
import { DEMO_CONNECTOR_ID, VSeedRender } from '@components'
import { useEffect, useState } from 'react'

export default () => {
  const [result, setResult] = useState<any>(null)

  useEffect(() => {
    const run = async () => {
      const builder = VBI.chart.create({
        ...{
          "connectorId": "demoSupermarket",
          "chartType": "table",
          "dimensions": [],
          "measures": [
            {
              "field": "sales",
              "alias": "원래 매출",
              "aggregate": {
                "func": "sum"
              },
              "encoding": "yAxis"
            }
          ],
          "whereFilter": {
            "id": "root",
            "op": "and",
            "conditions": []
          },
          "havingFilter": {
            "id": "root",
            "op": "and",
            "conditions": []
          },
          "theme": "light",
          "locale": "en-US",
          "version": 1,
          "limit": 20

    const measure = builder.measures.find((node) => node.getId() === measureId)
    if (measure) {
      measure.setAlias('조정할 매출').setEncoding('yAxis')
    }
    builder.measures.update(measureId, n => n.setAlias('새 매출').setAggregate({ func: 'avg' }))

}

      setResult(await builder.buildVSeed())
```
