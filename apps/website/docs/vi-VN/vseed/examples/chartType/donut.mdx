# Donut

## Basic Donut

A donut chart showing the proportional contribution of each category.

```tsx preview
import { VSeedRender } from '@components'

export default () => {
  const vseedConfig = {
  "chartType": "donut",
  "dataset": [
    {
      "date": "2019",
      "profit": 10,
      "sales": 20
    },
    {
      "date": "2020",
      "profit": 30,
      "sales": 60
    },
    {
      "date": "2021",
      "profit": 30,
      "sales": 60
    },
    {
      "date": "2022",
      "profit": 50,
      "sales": 100
    },
    {
      "date": "2023",
      "profit": 40,
      "sales": 80
    }
  ]
}

  return <VSeedRender vseed={vseedConfig} />
}
```

## Biểu đồ donut kết hợp

Biểu đồ donut kết hợp

```tsx preview
import { VSeedRender } from '@components'

export default () => {
  const vseedConfig = {
  "chartType": "donut",
  "measures": [
    {
      "id": "sales",
      "alias": "sales",
      "parentId": "sales-and-profit"
    },
    {
      "id": "profit",
      "alias": "Lợi nhuận",
      "parentId": "sales-and-profit"
    },
    {
      "id": "rateOfReturn",
      "alias": "Tỷ suất lợi nhuận",
      "parentId": "ratio"
    }
  ],
  "dataset": [
    {
      "date": "2019",
      "profit": 10,
      "sales": 20,
      "rateOfReturn": 0.1
    },
    {
      "date": "2020",
      "profit": 20,
      "sales": 40,
      "rateOfReturn": 0.2
    },
    {
      "date": "2021",
      "profit": 30,
      "sales": 60,
      "rateOfReturn": 0.3
    },
    {
      "date": "2022",
      "profit": 40,
      "sales": 80,
      "rateOfReturn": 0.4
    },
    {
      "date": "2023",
      "profit": 50,
      "sales": 100,
      "rateOfReturn": 0.5
    }
  ],
  "dimensions": [
    {
      "id": "date",
      "alias": "Ngày"
    }
  ]
}

  return <VSeedRender vseed={vseedConfig} />
}
```
